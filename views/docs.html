$url = 'https://www.reysekha.xyz/api/blabla';

curl_setopt($curl, CURLOPT_HTTPHEADER, [
 'Content-Type: application/json'
]);
$response = curl_exec($curl);
curl_close($curl);
echo $response . PHP_EOL; // appear response
